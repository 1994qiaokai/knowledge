6+1 中工程的解决办法。

if [ $? -ne 0 ];then 
	build/sbt  $SBT_MAVEN_PROFILES_ARGS  "test-only org.apache.spark.deploy.master.PersistenceEngineSuite"  | grep -v Resolving
fi
